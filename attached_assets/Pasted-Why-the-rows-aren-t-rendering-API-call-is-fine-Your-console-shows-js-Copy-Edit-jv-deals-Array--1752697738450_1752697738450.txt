arriving with HTTP 200, so the backend is delivering the 13 records.

Front-end parser mismatch – In dashboard.js (or your React hook) you’re doing something like:

js
Copy
Edit
const data  = await res.json();
populateDealsTable(data.deals);   //  <-- ❌ undefined
but the payload property is jv_deals, not deals.
When populateDealsTable(undefined) runs, undefined.forEach throws, the catch shows the TypeError, and DataTables receives zero rows.

🔧 2-line hot-fix
diff
Copy
Edit
- const { deals } = await res.json();
- populateDealsTable(deals);
+ const { jv_deals } = await res.json();
+ populateDealsTable(jv_deals ?? []);

populateDealsTable(Array.isArray(rows) ? rows : []);

async function loadDeals() {
  try {
    const res  = await fetch('/api/admin/jv-submissions');
    if (!res.ok) throw new Error(`HTTP ${res.status}`);
    const { jv_deals } = await res.json();

    populateDealsTable(jv_deals ?? []);
  } catch (err) {
    console.error('Error loading deals:', err);
    alert('Could not load JV submissions – check console for details.');
  }
}

function populateDealsTable(list) {
  // destroy old DataTable if it exists
  const $table = $('#jvDealsTable').DataTable();
  $table.clear();
  list.forEach(d => $table.row.add([
    d.partnerName,
    d.propertyAddress,
    d.location,
    formatMoney(d.askingPrice),
    formatMoney(d.suggestedOffer),
    statusBadge(d.status),
    actionBtns(d.id)
  ]));
  $table.draw();
}
