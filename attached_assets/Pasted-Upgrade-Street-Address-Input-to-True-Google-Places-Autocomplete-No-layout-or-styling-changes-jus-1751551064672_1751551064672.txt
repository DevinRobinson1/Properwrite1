Upgrade “Street Address” Input to True Google-Places Autocomplete

No layout or styling changes; just make sure the address comes back in the exact, canonical format Google Maps APIs expect and pre-populates the City, State, and ZIP boxes automatically.

2. Reusable Autocomplete Component
tsx
Copy
Edit
// components/GoogleAddressInput.tsx
import { usePlacesWidget } from "react-google-autocomplete";

export default function GoogleAddressInput({
  onSelect,
  defaultValue = ""
}) {
  const { ref } = usePlacesWidget({
    apiKey: process.env.GMAPS_KEY,
    onPlaceSelected: place => {
      const addr = place.address_components;
      const get = (type: string) =>
        addr.find(c => c.types.includes(type))?.long_name || "";

      onSelect({
        placeId: place.place_id,
        formatted: place.formatted_address,          // full Google format
        street: `${get("street_number")} ${get("route")}`,
        city: get("locality") || get("postal_town"),
        state: get("administrative_area_level_1"),
        zip: get("postal_code")
      });
    },
    types: ["address"],
    componentRestrictions: { country: "us" }
  });

  return (
    <input
      ref={ref}
      defaultValue={defaultValue}
      placeholder="Start typing address…"
      className="input input-bordered w-full"
      autoComplete="off"
    />
  );
}

<GoogleAddressInput

    setAddr(data);
    setCity(data.city);

    setZip(data.zip);
  }}

<input value={city}  onChange={e=>setCity(e.target.value)}  .../>
