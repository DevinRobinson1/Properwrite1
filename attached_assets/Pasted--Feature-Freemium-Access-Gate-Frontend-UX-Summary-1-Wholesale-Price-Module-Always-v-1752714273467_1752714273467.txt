	•	First click: allow it once, quietly set a cookie free_use=true and show a toast: “That was your free use—register now for more.”

	•	Seed user’s credit wallet with 5 free credits.
	•	Automatically close the modal and let them continue.

🧩 Backend Logic
	1.	Middleware: gateFeature(featureName)

import { getSession } from './auth.js';

export function gateFeature(featureName) {
  return async (req, res, next) => {
    const session = await getSession(req);
    if (session?.user) return next();           // Authenticated → OK

    // Publicly allow “wholesalePrice” feature
    if (featureName === 'wholesalePrice') return next();

    // Check free-use cookie
    const usedFree = req.cookies.free_use === 'true';
    if (!usedFree) {
      // First free use
      res.cookie('free_use', 'true', { maxAge: 365*24*3600*1000, httpOnly: false });
      return next();
    }

    // Otherwise block
    return res.status(403).json({ error: 'auth_required' });
  };
}

💻 Front-end Snippet (React)

// components/AuthGate.js
import { useState } from 'react';
import { useCookies } from 'react-cookie';
import SignupModal from './SignupModal';

export default function AuthGate({ feature, children }) {
  const [cookies, setCookie] = useCookies(['free_use']);
  const [showModal, setShowModal] = useState(false);

  const handleClick = async (e) => {
    // If logged-in, proceed
    if (window.__USER__) return children.props.onClick(e);

    // If this is the first free use
    if (!cookies.free_use) {
      setCookie('free_use','true',{ path:'/', maxAge:31536000 });
      // allow single use
      return children.props.onClick(e);
    }

    // otherwise show signup
    e.preventDefault();
    setShowModal(true);
  };

  // Clone the child (a button or link) and override onClick
  return (
    <>
      {React.cloneElement(children, { onClick: handleClick })}
      {showModal && <SignupModal onClose={()=>setShowModal(false)} />}
    </>
  );
}
