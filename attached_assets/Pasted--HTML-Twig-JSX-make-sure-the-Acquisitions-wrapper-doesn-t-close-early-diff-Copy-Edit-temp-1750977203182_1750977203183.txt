2. Tab Router Script (drop near bottom of page or in a JS bundle)

<script>
const components = {
  WHOLESALE: document.getElementById('wholesale-mao'),      // divs already in DOM
  NOVATION:  document.getElementById('novation-box'),
  SUB2:      document.getElementById('sub2-box'),
  SF:        document.getElementById('seller-finance-box'),
  COMPARE:   document.getElementById('compare-offers-box')
};

let active = "WHOLESALE";
render(active);

document.querySelectorAll('#acq-tabs .tab-bar li').forEach(li => {
  li.onclick = () => {
    active = li.dataset.tab;
    render(active);
  };
});

function render(key){
  const body = document.getElementById('acq-body');
  body.innerHTML = "";                       // empty previous
  Object.values(components).forEach(c => c && (c.style.display = 'none'));
  const el = components[key];
  if(el){
    el.style.display = 'block';
    body.appendChild(el);
  }
}
</script>
