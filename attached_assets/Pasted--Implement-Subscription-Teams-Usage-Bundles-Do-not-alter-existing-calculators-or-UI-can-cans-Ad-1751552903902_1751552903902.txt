  id uuid pk,
  team_id uuid,
  delta int,          -- +100 on purchase, -1 on analyze

);
3. Signup / Upgrade Flow
Choose plan → Stripe Checkout (pass seat quantity).

Webhook checkout.session.completed →
If new team: create teams row, owner user row.
If existing: update tier, seats_max.

Webhook invoice.paid: refresh credit_balance = included credits.

Webhook payment_intent.succeeded (one-time packs): credit_logs + credit_balance += packSize.

function requireSeat(req,res,next){
  const { team } = req.user;
  if(team.members.count > team.seats_max) return res.status(402).json({err:'Seat limit'});
  next();
