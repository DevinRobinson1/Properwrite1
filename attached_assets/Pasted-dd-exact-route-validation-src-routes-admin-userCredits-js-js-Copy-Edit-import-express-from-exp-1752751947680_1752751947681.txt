import express from 'express';
import { db } from '../../services/replitDb.js';   // ← adjust path
import { requireRole } from '../../middleware/requireRole.js';

const router = express.Router();

/**
 * POST /admin/api/users/:userId/credits
 * Body: { delta: number }
 */
router.post(
  '/admin/api/users/:userId/credits',
  requireRole(['admin', 'team']),
  express.json(),                       // <-- ensure JSON body is parsed
  async (req, res) => {
    try {
      const { userId } = req.params;
      const { delta } = req.body;

      if (!userId)          return res.status(400).json({ error: 'Missing userId' });
      if (typeof delta !== 'number' || isNaN(delta))
                           return res.status(400).json({ error: 'delta must be a number' });

      const subKey = `subscriptions:${userId}`;
      const sub    = (await db.get(subKey)) || { credits: 0 };

      const newCredits = Math.max(0, (sub.credits || 0) + delta);
      await db.set(subKey, { ...sub, credits: newCredits, updatedAt: Date.now() });

      res.json({ ok: true, credits: newCredits });
    } catch (err) {
      console.error('Credit adjust error', err);
      res.status(500).json({ error: 'Server error' });
    }
});

export default router;

async function addUserCredits(userId, delta) {
  try {
    const res = await fetch(`/admin/api/users/${userId}/credits`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      credentials: 'include',
      body: JSON.stringify({ delta: Number(delta) })
    });

    if (!res.ok) throw new Error(await res.text());
    const { credits } = await res.json();
    alert(`Credits updated ✔︎  New balance: ${credits}`);
    refreshUserPanel(userId);                          // re-pull user data
  } catch (err) {
    console.error('Error adding credits:', err);
    alert('Error adding credits');
  }
}
