/* components/AddressInput.jsx */
import { usePlacesWidget } from "react-google-autocomplete";
export default function AddressInput({ value, onSelect }) {
  const { ref } = usePlacesWidget({
    apiKey: process.env.GMAPS_KEY,
    onPlaceSelected: place => {
      onSelect({
        placeId: place.place_id,
        formatted: place.formatted_address
      });
    },
    types: ["address"],
    componentRestrictions: { country: "us" }
  });
  return <input ref={ref} defaultValue={value} placeholder="Street Address" />;
}
Drop-in replacement for every address box.

2. Server-Side Geocoding Fallback
js
Copy
Edit
// server/geocode.js
export async function geocodeLoose(text){
  const url = `https://maps.googleapis.com/maps/api/geocode/json?address=${encodeURIComponent(text)}&key=${GMAPS_KEY}`;
  const { results } = await fetch(url).then(r=>r.json());
  return results[0] ?? null;
}
If placeId missing, call geocodeLoose after POST, store highest-confidence result.

3. USPS / Smarty Normalization

Copy

import SmartyStreets from "@smartystreets/sdk";
const client = SmartyStreets.buildClient(...);

export async function normalizeAddress(addr){
  const [c] = await client.send({ street: addr, candidates:1 });
  return c ? c.deliveryLine1 + ", " + c.lastLine : null;
}
Save canonical_usps column; if null, flag submission for manual review.
